# EmojiChef Quick Recipe — Test Vectors
#
# Encoding parameters:
#   Base codepoint : 0x1F345 (🍅 tomato)
#   Max codepoint  : 0x1F37F (🍿 popcorn)
#   Bits per emoji : 6
#   Input alphabet : printable ASCII
#   Padding        : trailing incomplete 8-bit byte is dropped on decode
#
# Derivation:
#   encode(text) = split concat(charcode_binary(each char, 8 bits)) into 6-bit chunks
#                  → each chunk N maps to codepoint 0x1F345 + N
#   decode(emojis) = (codepoint - 0x1F345) per emoji → concat 6-bit strings
#                  → split into 8-bit groups → charcode to ASCII
#
# Format: <plaintext> TAB <encoded_emojis> TAB <emoji_count> TAB <note>
#
# IMPORTANT: Verify all vectors by running the encoder before using in tests.
# Quick Node verification:
#   node -e "
#     const B=0x1F345;
#     const enc=t=>[...t].map(c=>c.charCodeAt(0).toString(2).padStart(8,'0'))
#       .join('').match(/.{6}/g).map(s=>String.fromCodePoint(B+parseInt(s,2))).join('');
#     const dec=e=>[...e].map(c=>(c.codePointAt(0)-B).toString(2).padStart(6,'0'))
#       .join('').match(/.{8}/g).map(s=>String.fromCharCode(parseInt(s,2))).join('');
#     ['HSC','status','reboot'].forEach(t=>console.log(t,'->',enc(t),'->',dec(enc(t))));
#   "
