# EmojiChef Quick Recipe — Test Vectors
#
# Encoding parameters:
#   Base codepoint : 0x1F345 (🍅 tomato)
#   Max codepoint  : 0x1F384 (🎄 christmas tree)
#   Bits per emoji : 6
#   Range size     : 64 codepoints (offsets 0..63), covering every possible
#                    6-bit chunk value
#   Input alphabet : ASCII (any byte sequence; length 1..)
#   Padding        : encode pads input to the next multiple of 3 bytes with
#                    NUL (0x00); decode strips trailing NUL pad bytes
#
# Round-trip property: lossless for any ASCII input. Encoded output length
# is always Math.ceil(len(text)/3) * 4 emoji. Length-mod-3 inputs (e.g.
# "status", "reboot", "HSC") need no padding and produce byte-identical
# encodings to the pre-padding implementation. The range was extended past
# 🍿 (popcorn, offset 58) on 2026-05-06 to cover offsets 59..63 with the
# celebration-block emoji 🎀🎁🎂🎃🎄; before that, byte sequences whose
# bit-stream chunks happened to land in 59..63 produced emoji outside the
# documented range and the decoder threw on them. "reboot" was the most
# visible case (chunk 7 = 61 → 🎂); now it round-trips cleanly.
#
# Derivation:
#   encode(text):
#     padded = text + ('\0' * ((3 - len(text)%3) % 3))
#     bits   = concat(charcode_binary(c, 8 bits) for c in padded)
#     return  concat(String.fromCodePoint(0x1F345 + parseInt(b, 2))
#                    for b in chunks_of_6(bits))
#   decode(emojis):
#     bits   = concat(codepoint_minus_base_binary(e, 6 bits) for e in emojis)
#     bytes  = String.fromCharCode(parseInt(b, 2)) for b in chunks_of_8(bits)
#     return concat(bytes).rstrip('\0')
#
# Last verified : 2026-05-06 by running the Node snippet below
# Format        : <plaintext> TAB <encoded_emojis> TAB <emoji_count> TAB <note>
#
# Quick Node verification (re-run if the algorithm changes):
#   node -e "
#     const B=0x1F345;
#     const enc=t=>{
#       const pad=(3-t.length%3)%3, padded=t+'\0'.repeat(pad);
#       return [...padded].map(c=>c.charCodeAt(0).toString(2).padStart(8,'0'))
#         .join('').match(/.{6}/g).map(s=>String.fromCodePoint(B+parseInt(s,2))).join('');
#     };
#     const dec=e=>{
#       const bin=[...e].map(c=>(c.codePointAt(0)-B).toString(2).padStart(6,'0')).join('');
#       return (bin.match(/.{8}/g)||[]).map(s=>String.fromCharCode(parseInt(s,2)))
#         .join('').replace(/\0+\$/,'');
#     };
#     ['HSC','status','reboot','capture','list','ping','exec','fetch','capture 30']
#       .forEach(t=>console.log(t,'->',enc(t),'->',dec(enc(t))));
#   "
